DX = 150
DY = 150

VertexObject MyBaseQuad {
  @vertexCount(4)

  position: float32 {
    x [ -DX, DX, DX, -DX ]
    y [ DY, DY, -DY, -DY ]
    z
  }
}

Primitive TriQuads {
  @type(TRIANLGES)

  indices [
    0, 1, 2,
    0, 2, 3,
  ]
  stride 4
  offset 0
}

VertexObject MyQuads instantiates MyBaseQuad {
  translate {
    tx
    ty
    tz
  }
  color {
    r
    g
    b
    a
  }
  scale: uint8
  texCoords(size: 2): float32
  foo @uniform
  r(size: 1, offset: 3): float32 @alias
  uv @alias(texCoords)
}

SpriteGroup MySprites {
  @vertexObject(MyQuads)
  @primitive(TriQuads)
  @dynamic

  MyBaseQuads {
    @textured

    capacity 100

    texture {
      tex "/foo.png"
    }
  }

  MyQuads {
    @doublebuffer
  }
}
